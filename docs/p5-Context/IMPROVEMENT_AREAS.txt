name: DPR Validation

  validate:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v2
      - name: Run Validation
        run: python src/validate_standalone.py --section all
      - name: Check Threshold
        run: |
          if [ $SCORE -lt 90 ]; then
            echo "Score below threshold"
            exit 1
          fi
